Project 1: Music Generation with Python

In this project, I used Python to generate a simple melody automatically. This task shows that programming can also support creative work such as rhythm and melody generation. It helped me understand that Python is not only for problem-solving and automation, but also for artistic experimentation.

Music Generation Code

from mingus.containers import Bar, Track, Composition
from mingus.midi import midi_file_out
import random

notes_pool = ["C", "D", "E", "G", "A"]

bar = Bar()
bar.set_meter((4, 4))
bar.key = "C"

for _ in range(8):
    note_name = random.choice(notes_pool)
    bar.place_notes(note_name, 8)

track = Track()
track + bar

composition = Composition()
composition.add_track(track)

midi_file_out.write_Composition("melody.mid", composition)

print("MIDI file generated successfully: melody.mid")

This example creates a short MIDI melody automatically. It is a simple practice project to demonstrate music generation with Python.

Project 2: Video Processing with Python

In this project, I explored video editing with Python. The main idea is to detect silent parts in a video and remove them automatically. This function is useful because it can reduce repetitive manual editing and improve media production efficiency.

Video Processing Code

import moviepy.editor as mp
from pydub import AudioSegment
from pydub.silence import detect_silence
import os

def remove_silence(video_path, output_path="output.mp4", min_silence_len=500, silence_thresh=-40):
    video = mp.VideoFileClip(video_path)

    temp_audio_path = "temp_audio.wav"
    video.audio.write_audiofile(temp_audio_path, logger=None)

    audio_segment = AudioSegment.from_wav(temp_audio_path)

    silent_ranges = detect_silence(
        audio_segment,
        min_silence_len=min_silence_len,
        silence_thresh=silence_thresh
    )

    keep_ranges = []
    last_end = 0

    for start, end in silent_ranges:
        if start > last_end:
            keep_ranges.append((last_end / 1000, start / 1000))
        last_end = end

    if last_end < len(audio_segment):
        keep_ranges.append((last_end / 1000, len(audio_segment) / 1000))

    if not silent_ranges or not keep_ranges:
        video.write_videofile(output_path)
    else:
        clips = [video.subclip(start, end) for start, end in keep_ranges]
        final_clip = mp.concatenate_videoclips(clips)
        final_clip.write_videofile(output_path)
        final_clip.close()

    video.close()
    if os.path.exists(temp_audio_path):
        os.remove(temp_audio_path)

    print(f"Processed video saved as: {output_path}")

remove_silence("input.mp4", "edited_video.mp4")

This project is based on the idea of automatic silence removal. It demonstrates how Python can support practical video editing tasks in digital media.

Project 3: Mini Game Development with Pygame

In this project, I built a simple 2D mini game with Pygame. The player moves left and right to catch falling coins. This project helped me practice interaction design, game logic, keyboard control, and score counting.

Mini Game Code

import pygame
import random

pygame.init()

WIDTH, HEIGHT = 600, 400
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Catch the Coin")

WHITE = (255, 255, 255)
BLUE = (50, 100, 255)
GOLD = (255, 215, 0)
BLACK = (0, 0, 0)

player_width = 80
player_height = 20
player_x = WIDTH // 2 - player_width // 2
player_y = HEIGHT - 40
player_speed = 8

coin_radius = 12
coin_x = random.randint(20, WIDTH - 20)
coin_y = 0
coin_speed = 5

score = 0
font = pygame.font.SysFont(None, 36)

clock = pygame.time.Clock()
running = True

while running:
    clock.tick(30)
    screen.fill(WHITE)

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT] and player_x > 0:
        player_x -= player_speed
    if keys[pygame.K_RIGHT] and player_x < WIDTH - player_width:
        player_x += player_speed

    coin_y += coin_speed

    if (player_y < coin_y + coin_radius < player_y + player_height and
        player_x < coin_x < player_x + player_width):
        score += 1
        coin_x = random.randint(20, WIDTH - 20)
        coin_y = 0

    if coin_y > HEIGHT:
        coin_x = random.randint(20, WIDTH - 20)
        coin_y = 0

    pygame.draw.rect(screen, BLUE, (player_x, player_y, player_width, player_height))
    pygame.draw.circle(screen, GOLD, (coin_x, coin_y), coin_radius)

    score_text = font.render(f"Score: {score}", True, BLACK)
    screen.blit(score_text, (20, 20))

    pygame.display.flip()

pygame.quit()

This game is a beginner-friendly project, but it clearly shows how Python can be used to create interactive media experiences.
